import bpy import math # ============================== # CLEAR SCENE # ============================== bpy.ops.object.select_all(action='SELECT') bpy.ops.object.delete(use_global=False) # ============================== # THEME DEFINITIONS # ============================== THEMES = { "Christmas": { "cube_color": (0.8, 0.1, 0.1, 1.0), "ribbon_color": (1.0, 1.0, 1.0, 1.0), "glow_strength": 6.0 }, "Halloween": { "cube_color": (0.1, 0.1, 0.1, 1.0), "ribbon_color": (1.0, 0.4, 0.0, 1.0), "glow_strength": 5.0 }, "Birthday": { "cube_color": (0.2, 0.6, 1.0, 1.0), "ribbon_color": (1.0, 0.9, 0.4, 1.0), "glow_strength": 7.0 }, "Coffee": { "cube_color": (0.3, 0.2, 0.15, 1.0), "ribbon_color": (0.9, 0.8, 0.7, 1.0), "glow_strength": 4.0 } } ACTIVE_THEME = "Christmas" # ============================== # MATERIAL BUILDER (AR-SAFE) # ============================== def create_emission_material(name, color, strength): mat = bpy.data.materials.new(name) mat.use_nodes = True nodes = mat.node_tree.nodes links = mat.node_tree.links nodes.clear() output = nodes.new("ShaderNodeOutputMaterial") emission = nodes.new("ShaderNodeEmission") emission.inputs["Color"].default_value = color emission.inputs["Strength"].default_value = strength links.new(emission.outputs[0], output.inputs[0]) return mat # ============================== # CREATE GEOGIFT OBJECTS # ============================== # Core Cube bpy.ops.mesh.primitive_cube_add(size=0.3) cube = bpy.context.object cube.name = "GeoGift_Cube" cube.data.materials.append( create_emission_material( "Cube_Material", THEMES[ACTIVE_THEME]["cube_color"], 1.0 ) ) # Ribbon Loop bpy.ops.mesh.primitive_torus_add( major_radius=0.18, minor_radius=0.015, major_segments=48, minor_segments=16 ) ribbon = bpy.context.object ribbon.name = "Ribbon_Loop" ribbon.location.z = 0.17 ribbon.data.materials.append( create_emission_material( "Ribbon_Material", THEMES[ACTIVE_THEME]["ribbon_color"], 0.5 ) ) # AR Glow Ring bpy.ops.mesh.primitive_torus_add( major_radius=0.22, minor_radius=0.01, major_segments=64, minor_segments=12 ) ring = bpy.context.object ring.name = "AR_Glow_Ring" ring.location.z = -0.18 ring.data.materials.append( create_emission_material( "Glow_Material", (0.3, 1.0, 0.8, 1.0), THEMES[ACTIVE_THEME]["glow_strength"] ) ) # ============================== # HOVER ANIMATION (IDLE AR LOOP) # ============================== for obj in [cube, ribbon, ring]: obj.animation_data_create() action = bpy.data.actions.new(name=f"{obj.name}_Hover") obj.animation_data.action = action fcurve = action.fcurves.new(data_path="location", index=2) fcurve.keyframe_points.add(2) base_z = obj.location.z fcurve.keyframe_points[0].co = (1, base_z) fcurve.keyframe_points[1].co = (60, base_z + 0.03) for kp in fcurve.keyframe_points: kp.interpolation = 'SINE' print("GeoGift Parametric Generator executed successfully.") print("Active Theme:", ACTIVE_THEME)
